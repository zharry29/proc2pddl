(:action gather_info
   :parameters (?p - player)
   :precondition (and (not (has_info ?p)) (society_is_ready ?sc))
   :effect (has_info ?p)
)

(:action create_secret
   :parameters (?sc - secret_society)
   :precondition (society_is_ready ?sc)
   :effect (has_secret ?sc)
)

(:action set_meeting_location
   :parameters (?sc - secret_society)
   :precondition (society_is_ready ?sc)
   :effect (has_location ?sc)
)

(:action set_dress_code
   :parameters (?sc - secret_society)
   :precondition (society_is_ready ?sc)
   :effect (has_dress_code ?sc)
)

(:action create_name
   :parameters (?sc - secret_society)
   :precondition (society_is_ready ?sc)
   :effect (has_name ?sc)
)

(:action secret_society_is_ready
   :parameters (?sc - secret_society)
   :precondition (and (not (has_secret ?sc)) (has_location ?sc) (has_dress_code ?sc) (has_name ?sc))
   :effect (society_is_ready ?sc)
)

(:action initiate_new_member
   :parameters (?p - player ?sc - secret_society)
   :precondition (and (society_is_ready ?sc) (not (in ?p ?sc)) (is_trusted ?p ?sc))
   :effect (in ?p ?sc)
)

(:action teach_new_member_secret
   :parameters (?p - player ?sc - secret_society)
   :precondition (and (in ?p ?sc) (is_trusted ?p ?sc) (has_secret ?sc))
   :effect (knows_secret ?p ?sc)
)

(:action remove_member
   :parameters (?p - player ?sc - secret_society)
   :precondition (and (in ?p ?sc) (not (is_trusted ?p ?sc)))
   :effect (not (in ?p ?sc))
)

(:action check_if_cult
   :parameters (?sc - secret_society)
   :precondition (and (society_is_ready ?sc) (>= (count ?p - player :condition (in ?p ?sc)) 3))
   :effect (is_cult ?sc)
)

)
