(:action go
    (?player - player ?from - location ?dir - direction ?to - location)
    (and
      (at ?player ?from)
      (connected ?from ?dir ?to)
      (not (blocked ?from ?dir ?to))
    )
    (and
      (not (at ?player ?from))
      (at ?player ?to)
    )
)

(:action get
    (?player - player ?item - item ?loc - location)
    (and
      (at ?player ?loc)
      (at ?item ?loc)
    )
    (and
      (not (at ?item ?loc))
      (inventory ?player ?item)
    )
)

(:action get_water
    (?player - player ?loc - location ?water - water)
    (and
      (at ?player ?loc)
      (has_water_source ?loc)
      (at ?water ?loc)
    )
    (and
      (not (at ?water ?loc))
      (inventory ?player ?water)
    )
)

(:action sharpen_stick
    (?player - player ?stick - stick ?sharpened_stick - stick)
    (and
      (inventory ?player ?stick)
      (not (inventory ?player ?sharpened_stick))
    )
    (and
      (not (inventory ?player ?stick))
      (inventory ?player ?sharpened_stick)
    )
)

(:action catch_fish
    (?player - player ?loc - location ?sharpened_stick - stick ?fish - fish)
    (and
      (at ?player ?loc)
      (inventory ?player ?sharpened_stick)
      (has_fish ?loc)
    )
    (and
      (not (has_fish ?loc))
      (inventory ?player ?fish)
      (edible ?fish)
    )
)
