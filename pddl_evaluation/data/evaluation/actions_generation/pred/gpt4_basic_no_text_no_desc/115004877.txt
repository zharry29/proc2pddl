(:action go
    :parameters (?player - player ?dir - direction)
    :precondition (and (at ?player ?loc1 - location) (connected ?loc1 ?dir ?loc2 - location))
    :effect (and (not (at ?player ?loc1)) (at ?player ?loc2)))

(:action get
    :parameters (?player - player ?item - item)
    :precondition (and (at ?item ?loc) (at ?player ?loc) (not (inventory ?player ?item)))
    :effect (inventory ?player ?item))

(:action cut_plant
    :parameters (?player - player ?scissors - scissors ?plant - plant ?stalk - stalk)
    :precondition (and (inventory ?player ?scissors) (at ?player ?loc) (at ?plant ?loc))
    :effect (not (at ?plant ?loc)) (at ?stalk ?loc))

(:action peel_stalk
    :parameters (?player - player ?stalk - stalk)
    :precondition (and (inventory ?player ?stalk) (at ?player ?loc))
    :effect (peeled ?stalk))

(:action cut_stalk_into_strips
    :parameters (?player - player ?stalk - stalk ?strips - strip)
    :precondition (and (peeled ?stalk) (at ?player ?loc) (inventory ?player ?stalk))
    :effect (not (at ?player ?stalk)) (inventory ?strips))

(:action soak_strips
    :parameters (?player - player ?strips - strip ?loc - location)
    :precondition (and (at ?player ?loc) (has_water ?loc) (inventory ?strips))
    :effect (clean ?strips))

(:action dry_out_strips
    :parameters (?player - player ?strips - strip)
    :precondition (and (inventory ?player ?strips) (clean ?strips))
    :effect (dry ?strips))

(:action weave_strips_into_sheet
    :parameters (?player - player ?sheet - sheet ?strips - strip)
    :precondition (and (inventory ?player ?strips) (dry ?strips))
    :effect (at ?sheet ?loc))

(:action press_sheet
    :parameters (?player - player ?sheet - sheet ?board - wooden_board)
    :precondition (and (at ?player ?loc) (has_hard_surface ?loc) (inventory ?player ?sheet) (inventory ?player ?board))
    :effect (flat ?sheet))

(:action flatten_sheet
    :parameters (?player - player ?sheet - sheet)
    :precondition (and (at ?player ?loc) (has_hard_surface ?loc) (flat ?sheet) (inventory ?player ?sheet))
    :effect (very_flat ?sheet))

(:action polish_sheet
    :parameters (?player - player ?sheet - sheet ?shell - shell)
    :precondition (and (inventory ?player ?sheet) (very_flat ?sheet) (inventory ?player ?shell))
    :effect (smooth ?sheet))

(:action cut_sheet
    :parameters (?player - player ?sheet - sheet ?scissors - scissors ?papyrus - papyrus)
    :precondition (and (smooth ?sheet) (inventory ?player ?sheet) (inventory ?player ?scissors))
    :effect (and (not (at ?player ?sheet)) (inventory ?player ?papyrus)))
