(:action go
        :parameters (?player ?dir - direction ?loc1 - location ?loc2 - location)
        :precondition (and (not (blocked ?loc1 ?dir ?loc2)) 
                           (at ?player ?loc1) 
                           (connected ?loc1 ?dir ?loc2))
        :effect (and (not (at ?player ?loc1)) 
                     (at ?player ?loc2))
    )

    (:action get
        :parameters (?player ?item - item ?loc - location)
        :precondition (and (at ?player ?loc) 
                           (at ?item ?loc))
        :effect (and (not (at ?item ?loc)) 
                     (inventory ?player ?item))
    )

    (:action get_water
        :parameters (?player ?water - water ?loc - location)
        :precondition (and (at ?player ?loc) 
                           (has_water ?loc))
        :effect (and (not (has_water ?loc)) 
                     (inventory ?player ?water))
    )

    (:action bag_plant
        :parameters (?player ?plant - plant ?bag - container ?loc - location)
        :precondition (and (at ?plant ?loc) 
                           (at ?player ?loc) 
                           (inventory ?player ?bag))
        :effect (bagged ?plant)
    )

    (:action wait_and_gather_bagged_plant
        :parameters (?player ?plant - plant ?bag - container)
        :precondition (and (bagged ?plant) 
                           (inventory ?player ?bag))
        :effect (inventory ?player ?plant)
    )

    (:action gather_tinder
        :parameters (?player ?tinder - tinder ?loc - location)
        :precondition (and (at ?player ?loc) 
                           (has_tinder ?loc))
        :effect (and (not (has_tinder ?loc)) 
                     (inventory ?player ?tinder))
    )

    (:action build_campfire
        :parameters (?player ?loc - location ?wood - wood)
        :precondition (and (at ?player ?loc) 
                           (inventory ?player ?wood))
        :effect (and (not (inventory ?player ?wood)) 
                     (at ?wood ?loc))
    )

    (:action light_campfire
        :parameters (?player ?campfire - campfire ?loc - location ?tinder - tinder)
        :precondition (and (at ?player ?loc) 
                           (inventory ?player ?tinder) 
                           (at ?campfire ?loc))
        :effect (lit ?campfire)
    )

    (:action make_spear
        :parameters (?player ?sharp_stone - sharp_stone)
        :precondition (inventory ?player ?sharp_stone)
        :effect (inventory ?player spear)
    )

    (:action spear_fish
        :parameters (?player ?fish - fish ?loc - location ?spear - spear)
        :precondition (and (at ?player ?loc) 
                           (has_fish ?loc) 
                           (inventory ?player ?spear))
        :effect (and (not (has_fish ?loc)) 
                     (inventory ?player ?fish))
    )

    (:action cook_fish
        :parameters (?player ?fish - fish ?campfire - campfire ?loc - location)
        :precondition (and (at ?player ?loc) 
                           (lit ?campfire) 
                           (at ?fish ?loc))
        :effect (cooked ?fish)
    )

    (:action wet_tinder
        :parameters (?player ?tinder - tinder ?water - water)
        :precondition (and (inventory ?player ?tinder) 
                           (inventory ?player ?water))
        :effect (wet ?tinder)
    )

    (:action make_smoke_signal
        :parameters (?player ?campfire - campfire ?tinder - tinder ?loc - location)
        :precondition (and (at ?player ?loc) 
                           (lit ?campfire) 
                           (at ?campfire ?loc)
                           (wet ?tinder))
        :effect (signaling ?campfire)
    )

)
