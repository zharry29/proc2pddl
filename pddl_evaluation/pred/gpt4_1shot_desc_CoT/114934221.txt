(:action splash_swim
 :parameters (?player ?loc - location ?dir - direction ?loc2 - location)
 :precondition (and (at ?player ?loc) (is_water ?loc) (connected ?loc ?dir ?loc2) (not (defending ?player)))
 :effect (and (not (at ?player ?loc)) (at ?player ?loc2) (wounded ?player)))

(:action swim
 :parameters (?player ?loc - location ?dir - direction ?loc2 - location)
 :precondition (and (at ?player ?loc) (is_water ?loc) (wounded ?player) (connected ?loc ?dir ?loc2))
 :effect (and (not (at ?player ?loc)) (at ?player ?loc2)))

(:action get
 :parameters (?player ?item - body)
 :precondition (and (visible ?item) (not (inventory ?player ?item)))
 :effect (inventory ?player ?item))

(:action onboard
 :parameters (?player ?ship)
 :precondition (and (visible ?ship) (at ?player ?loc - location) (on ?ship ?loc))
 :effect (and (on ?player ?ship) (not (at ?player ?loc))))

(:action defend
 :parameters (?player)
 :precondition (not (defending ?player))
 :effect (defending ?player))

(:action spot_ship
 :parameters (?player ?ship)
 :precondition (and (not (visible ?ship)) (at ?player ?loc - location) (on ?ship ?loc))
 :effect (visible ?ship))

(:action spot_shark
 :parameters (?player ?shark)
 :precondition (and (not (visible ?shark)) (at ?player ?loc - location) (at ?shark ?loc))
 :effect (visible ?shark))

(:action self_mobilize
 :parameters (?player ?shark)
 :precondition (and (at ?player ?loc - location) (at ?shark ?loc) (not (wounded ?player)))
 :effect (and (wounded ?player) (attacked ?shark ?player)))

(:action medicare
 :parameters (?player)
 :precondition (wounded ?player)
 :effect (not (wounded ?player)))

(:action attack_vulnerable
 :parameters (?player ?shark ?obj - object)
 :precondition (and (vulnerable ?part) (sharp ?obj) (inventory ?player ?obj) (visible ?shark))
 :effect (attacked ?player ?shark))

(:action knockout
 :parameters (?player ?shark ?obj - object)
 :precondition (and (vulnerable ?part) (heavy ?obj) (inventory ?player ?obj) (visible ?shark))
 :effect (and (attacked ?player ?shark) (dizzy ?shark)))
